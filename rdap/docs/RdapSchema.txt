	"ID" INTEGER NOT NULL,

	"DRIVER_ID" INTEGER NOT NULL,
	"VIOLATION_ID" INTEGER NOT NULL,
	PRIMARY KEY ( "ID" ),
	CONSTRAINT "DRIVER_FK" FOREIGN KEY ( "DRIVER_ID" ) REFERENCES "DRIVER" ( "ID" ),
